import React, { useState, useEffect } from 'react';
import * as api from './api';

function App() {

  const companyName = "My New Company Manager";

  const [companies, setCompanies] = useState([]);
  const [formData, setFormData] = useState({ company_name: '', employees: 0, location: '', is_active: true });
  const [editingId, setEditingId] = useState(null);

  // Load companies on mount
  useEffect(() => {
    fetchCompanies();
  }, []);

  const fetchCompanies = async () => {
    try {
      const response = await api.getCompanies();
      setCompanies(response.data);
    } catch (error) {
      console.error("Error fetching companies:", error);
    }
  };

  const handleSubmit = async (e) => {
    e.preventDefault();
    try {
      if (editingId) {
        await api.updateCompany(editingId, formData);
      } else {
        await api.createCompany(formData);
      }
      setFormData({ company_name: '', employees: 0, location: '', is_active: true });
      setEditingId(null);
      fetchCompanies();
    } catch (error) {
      console.error("Error saving company:", error);
    }
  };

  const handleEdit = (company) => {
    setEditingId(company.company_id);
    setFormData({
      company_name: company.company_name,
      employees: company.employees,
      location: company.location,
      is_active: company.is_active
    });
  };

  const handleDelete = async (id) => {
    if (window.confirm("Are you sure you want to delete this company?")) {
      await api.deleteCompany(id);
      fetchCompanies();
    }
  };

  const getCompanyName = (cName) => {
    return "Hello this is " + cName + " Company";
  }

  return (
    <div style={{ padding: '20px', fontFamily: 'Arial' }}>
      <h1>{getCompanyName('Kumar')}</h1>

      {/* Form Section */}
      <form onSubmit={handleSubmit} style={{ marginBottom: '30px', border: '1px solid #ccc', padding: '15px' }}>
        <h3>{editingId ? 'Update Company' : 'Add New Company'}</h3>
        <input 
          type="text" placeholder="Company Name" required
          value={formData.company_name}
          onChange={(e) => setFormData({...formData, company_name: e.target.value})}
        />
        <input 
          type="number" placeholder="Employees" required
          value={formData.employees}
          onChange={(e) => setFormData({...formData, employees: parseInt(e.target.value)})}
        />
        <input 
          type="text" placeholder="Location" required
          value={formData.location}
          onChange={(e) => setFormData({...formData, location: e.target.value})}
        />
        <input 
          type="checkbox" id="active" checked={formData.is_active}
          onChange={(e) => setFormData({...formData, is_active: e.target.checked})}
        />
        <label htmlFor="active">Active</label>

        <button type="submit">{editingId ? 'Update' : 'Create'}</button>
        {editingId && <button onClick={() => {setEditingId(null); setFormData({company_name:'', employees:0, location:''})}}>Cancel</button>}
      </form>

      {/* List Section */}
      <table border="1" cellPadding="10" style={{ width: '100%', textAlign: 'left' }}>
        <thead>
          <tr>
            <th>Name</th>
            <th>Employees</th>
            <th>Location</th>
            <th>Active</th>
            <th>Actions</th>
          </tr>
        </thead>
        <tbody>
          {
            companies.map(company => (
              <tr key={company.company_id}>
                <td>{company.company_name}</td>
                <td>{company.employees}</td>
                <td>{company.location}</td>
                <td>{company.is_active ? '✅' : '❌'}</td>
                <td>
                  <button onClick={() => handleEdit(company)}>Edit</button>
                  <button onClick={() => handleDelete(company.company_id)}>Delete</button>
                </td>
              </tr>
            ))
          }
        </tbody>
      </table>
    </div>
  );
}

export default App;
